\begin{table}[p]
\caption{\textbf{Comparative features of oncology variant knowledge bases.} OncoKB, COSMIC, and JAX-CKB require institutional licenses. OncoKB aligns to AMP/ASCO/CAP standards; ClinVar follows ACMG/AMP frameworks. Additional details in Supplementary Table~S1.}\label{tab1}
\centering
\begin{tabular}{@{}lcccc@{}}
\toprule
\textbf{Feature} & \textbf{CIViC} & \textbf{OncoKB} & \textbf{COSMIC} & \textbf{ClinVar} \\
\midrule
Access model & Open & License & License & Open \\
API support & Yes & Yes & Yes & Yes \\
Natural language query & Partial & No & No & No \\
Evidence standards & Dual$^{\mathrm{a}}$ & AMP/CAP & Mixed & ACMG \\
JSON export & Yes & License & License & Yes \\
Gene normalization & Yes & Partial & Partial & Partial \\
EHR integration & Partial & Partial & No & Partial \\
\botrule
\end{tabular}

\medskip
\raggedright\small
$^{\mathrm{a}}$CIViC uses dual A--E evidence levels plus 1--5 star trust ratings.
\end{table}
